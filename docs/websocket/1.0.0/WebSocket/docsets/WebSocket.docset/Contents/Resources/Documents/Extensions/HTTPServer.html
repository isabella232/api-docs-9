HTTPServer

Allows HTTPServer to accept WebSocket connections.

    // return non-nil HTTPHeaders to allow upgrade
}, onUpgrade: { ws, req in
    // setup callbacks or send data to connected WebSocket
})

HTTPServer.start(..., upgraders: [ws])
